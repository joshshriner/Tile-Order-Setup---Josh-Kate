      <div class="step-title">Enter Quantities</div>

      <div class="step-sub">Fill in what you need. Skipped items are excluded.</div>

    </div>

  `;

  rooms.forEach(room => {

    const active = room.items.filter(i => !edits[i.id].skipped);

    if (!active.length) return;

    html += `<div class="room-header">${room.name}</div>`;

  `;

}

  let summaryRows = '';

  rooms.forEach(room => {

    const orderItems = room.items.filter(i => !edits[i.id].skipped && quantities[i.id] !== '' && Number(quantities[i.id]) > 0);

    if (!orderItems.length) return;

    hasItems = true;

    summaryRows += `<div class="summary-room">${room.name}</div>`;

    orderItems.forEach(item => {

      const e = edits[item.id];

      summaryRows += `

        <div class="summary-row">

          <div>

            <div class="summary-item-name">${item.label} — ${e.name}</div>

            <div class="summary-item-sku">SKU: ${e.sku}${e.desc?' · '+e.desc:''}</div>

          </div>

          <div class="summary-qty">${quantities[item.id]} ${item.unit}</div>

        </div>

      `;

    });

  });


  let html = `

    <div class="step-header">

      <div class="step-label">PHASE 3 — SUMMARY</div>

      <div class="step-title">Order Review</div>

      <div class="step-sub">Review, add notes, then copy or download as PDF.</div>

    </div>

    <div class="note-box">

      <label>Project / Address</label>

      <input type="text" placeholder="e.g. 501 S 24th St Broken Arrow" value="${projectName}" oninput="projectName=this.value" />

    </div>

    ${hasItems ? summaryRows : '<div class="empty-state">No items with quantities entered.<br>Go back and add some.</div>'}

    <div class="note-box" style="margin-top:16px">

    ` : ''}

function copyOrder() {

  const date = new Date().toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });

  const lines = ['WILDE HOLDINGS — TILE ORDER'];

  if (projectName) lines.push('PROJECT: ' + projectName);

  lines.push('DATE: ' + date, '');

  rooms.forEach(room => {

    const items = room.items.filter(i => !edits[i.id].skipped && quantities[i.id] !== '' && Number(quantities[i.id]) > 0);

    if (!items.length) return;

    lines.push('── ' + room.name.toUpperCase() + ' ──');

    items.forEach(item => {

      const e = edits[item.id];

      lines.push(item.label + ': ' + e.name + (e.desc ? ' (' + e.desc + ')' : ''));

      lines.push('  SKU: ' + e.sku + '  |  Qty: ' + quantities[item.id] + ' ' + item.unit);

    });

    lines.push('');

  });

  if (notes.trim()) lines.push('NOTES: ' + notes.trim());

  const text = lines.join('\n');

  navigator.clipboard.writeText(text).catch(() => {

    const ta = document.createElement('textarea');

    ta.value = text; document.body.appendChild(ta); ta.select();

    document.execCommand('copy'); document.body.removeChild(ta);

  });

  copied = true; render();

  setTimeout(() => { copied = false; render(); }, 3000);

}


function generatePDF() {

  const date     = new Date().toLocaleDateString('en-US', { month: 'long',  day: 'numeric', year: 'numeric' });

  const dateShort= new Date().toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' });

  const unitLabel = u => u === 'sqft' ? 'sq ft' : u === 'bags' ? 'bags' : 'pcs';


  let roomsHtml = '';

  rooms.forEach(room => {

    const items = room.items.filter(i => !edits[i.id].skipped && quantities[i.id] !== '' && Number(quantities[i.id]) > 0);

    if (!items.length) return;

    roomsHtml += `<div class="pf-room-label">${room.name}</div>`;

    roomsHtml += `

      <table class="pf-table">

        <thead>

          <tr>

            <th style="width:26%">Item</th>

            <th style="width:36%">Product</th>

            <th style="width:18%">SKU</th>

            <th class="r" style="width:20%">Qty</th>

          </tr>

        </thead>

        <tbody>

          ${items.map((item, idx) => {

            const e = edits[item.id];

            return `

              <tr class="${idx % 2 === 1 ? 'even' : ''}">

                <td><div class="pf-item-label">${item.label}</div></td>

                <td>

                  <div>${e.name}</div>

                  ${e.desc ? `<div class="pf-item-desc">${e.desc}</div>` : ''}

                </td>

                <td><span class="pf-sku">${e.sku}</span></td>

                <td class="pf-qty">${quantities[item.id]} <span class="pf-unit">${unitLabel(item.unit)}</span></td>

              </tr>

            `;

          }).join('')}

        </tbody>

      </table>

    `;

  });


  document.getElementById('print-frame').innerHTML = `

    <div class="pf-header">

      <div>

        <div class="pf-company">Wilde Holdings, LLC</div>

        <div class="pf-tagline">Tile Material Order</div>

      </div>

      <div class="pf-meta">

        <div class="pf-meta-label">Date</div>

        <div class="pf-meta-value">${date}</div>

      </div>

    </div>

    ${projectName ? `

      <div class="pf-project">

        <div class="pf-project-label">Project</div>

        <div class="pf-project-name">${projectName}</div>

      </div>` : ''}

    ${roomsHtml}

    ${notes.trim() ? `

      <div class="pf-notes">

        <div class="pf-notes-label">Notes</div>

        <div class="pf-notes-text">${notes.trim()}</div>

      </div>` : ''}

    <div class="pf-footer">

      <span>Wilde Holdings, LLC — Tile Order</span>

      <span>${dateShort}${projectName ? ' · ' + projectName : ''}</span>

    </div>

  `;


  const oldTitle = document.title;

  document.title = 'Tile Order' + (projectName ? ' — ' + projectName : '') + ' — ' + dateShort;

  window.print();

  setTimeout(() => { document.title = oldTitle; }, 1500);

}

}
